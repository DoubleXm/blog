
## createParamDecorator

自定义参数装饰器，常规的 `Controller` 中获取接口数据的装饰器，都具有单一功能，有时可能需要获取一些更个性的数据，需要用到它。

接受两个参数 `data` 则是传递的参数， `context` 就是拦截器中的 `ExecutionContext` 对象

```ts :no-line-numbers
import { createParamDecorator, ExecutionContext } from '@nestjs/common';

export User = createParamDecorator(
  data: string,
  context: ExecutionContext
) {
  const req = context.switchToHttp().getRequest();
  const user = req.user;

  // 尝试获取请求头的 user 中指定的值
  return data ? user?.[data] : user;
}

// users.controller.ts
@Get()
async findOne(@User('firstName') firstName: string) {
  console.log(`Hello ${firstName}`);
}

// 也可以与管道一起使用但是 validateCustomDecorators 一定要传递为 true
@Get()
async findOne(
  @User(new ValidationPipe({ validateCustomDecorators: true }))
  user: UserEntity,
) {
  console.log(user);
}
```

## applyDecorators

可以将多个装饰器组装成一个装饰器，增强代码的可读性和复用性

```ts :no-line-numbers
import { applyDecorators } from '@nestjs/common';

// 参数就是装饰器的参数
export function Auth(...roles: Role[]) {
  return applyDecorators(
    decorator1,
    decorator2,
    ...
  )
}
```

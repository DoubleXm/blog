## 库的 CURD

```sql
show databases; -- 查看所有数据库
create database [库名]; -- 创建数据库
show create database [库名]; -- 查看数据库创建语句
alter database [库名] charset=utf8; -- 修改数据库编码
drop database [库名]; -- 删除数据库
use [库名]; -- 进入某个数据库
select database(); -- 查看当前所在数据库
```

## 查看表语法

```sql
show create table [表名]; -- 查看表的创建语法
-- 查看表结构
describe [表名];
desc [表名];
show tables; -- 查看所有表
```

## 创建表的语法

```sql
create table <表名> (
  <字段名1> <字段类型> [(宽度)] [约束条件1 约束条件2],
  <字段名2> <字段类型> [(宽度)] [约束条件]
);

-- 示例
create table movies (
  id int not null auto_increment, -- 自增且不为空
  name char(10) not null, -- 不能为空
  age int default 1, -- 默认值 1
  primary key (id) -- 将 id 设置为主键
);
```

## 修改表的语法

```sql 
-- 修改存储引擎
alter table [表名] engine=<存储引擎名字>;

-- 修改表名
alter table [表名] rename <新表名>;

-- 增加字段 first 第一个；after 某个字段后面
alter table [表名] 
  add <字段名><字段类型> [约束条件] [first | after<字段名>],
  add 字段2...;

-- 删除字段
alter table [表名] drop <字段名>;

-- 修改字段类型
alter table [表名] modify [字段名]<字段类型> [约束条件]

-- 修改字段名及类型
alter table [表名] change [旧字段名] [新字段名]<新字段类型> [约束条件]
```


## 删除和复制表语法

```sql
-- 删除表
drop table [表名]

-- 复制表
create table [表名] select * from <被复制表名> [条件];

-- 复制表结构
create table [表名] like <被复制表名>;
```
